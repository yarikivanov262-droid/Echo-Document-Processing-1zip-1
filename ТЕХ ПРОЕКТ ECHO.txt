================================================================================
         ECHO MESSENGER — ПОЛНЫЙ ТЕХНИЧЕСКИЙ ПРОЕКТ v2.0
         «Говори. Никто не узнает.»
         Аналог Telegram. Без телефона. Без email. Только seed-фраза.
================================================================================

СТЕК:
  Node.js 24 + Express 5 + PostgreSQL + Drizzle ORM
  React 18 + Vite + Tailwind CSS + Framer Motion + Wouter + TanStack Query
  WebSocket (ws lib) + WebRTC (нативный браузер API)
  Монорепо pnpm workspaces

СТРУКТУРА МОНОРЕПО:
  artifacts/api-server/         — Express backend (порт 8080)
  artifacts/echo-messenger/     — React PWA frontend (порт 5000)
  lib/db/src/schema/            — Drizzle ORM таблицы
  lib/api-spec/openapi.yaml     — OpenAPI контракт (источник истины)
  lib/api-client-react/         — Сгенерированные TanStack Query хуки
  lib/api-zod/                  — Сгенерированные Zod-схемы

=== STARS & PREMIUM API (создать artifacts/api-server/src/routes/stars.ts) ===

GET /api/stars/balance
  SELECT stars_balance FROM users WHERE id = userId
  Ответ: { balance: number }

GET /api/stars/transactions
  SELECT * FROM stars_transactions WHERE from_user_id = userId OR to_user_id = userId ORDER BY created_at DESC
  Ответ: [{ id, type, amount, description, fromUserId, toUserId, createdAt }]

POST /api/stars/gift
  Body: { toUserId: number, amount: number }
  Проверить что users.starsBalance >= amount
  BEGIN TRANSACTION:
    UPDATE users SET stars_balance = stars_balance - amount WHERE id = userId
    UPDATE users SET stars_balance = stars_balance + amount WHERE id = toUserId
    INSERT INTO stars_transactions (from, to, amount, type: "gift")
  COMMIT
  Broadcast WS к toUserId: { type: "stars_received", amount, fromUsername }
  Ответ: { success: true, newBalance: number }

POST /api/stars/purchase
  Body: { package: "100" | "250" | "500" | "1000" | "2500" }
  Пакеты цен (демо-режим, без реальной оплаты):
    "100" → +100 stars, стоимость 0 (демо)
    Реальная интеграция: проверить payment_id от платёжного провайдера
  UPDATE users SET stars_balance = stars_balance + :amount WHERE id = userId
  INSERT INTO stars_transactions (to_user_id: userId, amount, type: "purchase")
  Ответ: { success: true, newBalance: number }

POST /api/stars/tip
  Body: { messageId: number, amount: number }
  Найти senderId сообщения
  Списать stars у текущего пользователя, начислить автору
  INSERT INTO reactions_paid (message_id, user_id, stars_amount)
  Broadcast WS к chatId: { type: "paid_reaction", messageId, amount, fromUsername }
  Ответ: { success: true }

POST /api/premium/subscribe
  Body: { tier: "monthly" | "annual" }
  Цены: monthly = 280 Stars, annual = 2800 Stars
  Проверить баланс, списать Stars
  UPDATE users SET is_premium = true, premium_until = :date WHERE id = userId
  INSERT INTO premium_subscriptions
  Ответ: { success: true, expiresAt: string }

  import { createContext, useContext, useState, useCallback } from "react";
  import { useWsEvent } from "@/hooks/use-ws";

  type OnlineMap = Record<number, { isOnline: boolean; lastOnline: string | null }>;
  const OnlineCtx = createContext<OnlineMap>({});
  const OnlineSetCtx = createContext<(userId: number, data: OnlineMap[number]) => void>(() => {});

  export function OnlineProvider({ children }: { children: React.ReactNode }) {
    const [map, setMap] = useState<OnlineMap>({});
    const update = useCallback((userId: number, data: OnlineMap[number]) => {
      setMap(prev => ({ ...prev, [userId]: data }));
    }, []);

    useWsEvent((event) => {
      if (event.type === "status") {
        update(event.userId, { isOnline: event.online, lastOnline: event.lastOnline ?? null });
      }
    });

    return (
      <OnlineCtx.Provider value={map}>
        <OnlineSetCtx.Provider value={update}>
          {children}
        </OnlineSetCtx.Provider>
      </OnlineCtx.Provider>
    );
  }

  export const useOnlineStatus = (userId?: number) => {
    const map = useContext(OnlineCtx);
    if (!userId) return null;
    return map[userId] ?? null;
  };

  export function useVoiceRecorder() {
    const [isRecording, setIsRecording] = useState(false);
    const [duration, setDuration] = useState(0);
    const [waveform, setWaveform] = useState<number[]>([]);
    const mediaRecorderRef = useRef<MediaRecorder | null>(null);
    const chunksRef = useRef<Blob[]>([]);
    const timerRef = useRef<number | null>(null);
    const analyserRef = useRef<AnalyserNode | null>(null);
    const animFrameRef = useRef<number | null>(null);

    const startRecording = async () => {
      const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
      
      // Waveform analyser
      const audioCtx = new AudioContext();
      const source = audioCtx.createMediaStreamSource(stream);
      const analyser = audioCtx.createAnalyser();
      analyser.fftSize = 64;
      source.connect(analyser);
      analyserRef.current = analyser;
      
      const recorder = new MediaRecorder(stream, { mimeType: "audio/webm;codecs=opus" });
      chunksRef.current = [];
      recorder.ondataavailable = e => { if (e.data.size > 0) chunksRef.current.push(e.data); };
      recorder.start(100);
      mediaRecorderRef.current = recorder;
      setIsRecording(true);
      setDuration(0);
      timerRef.current = window.setInterval(() => setDuration(d => d + 1), 1000);
      
      // Update waveform
      const updateWave = () => {
        const data = new Uint8Array(analyser.frequencyBinCount);
        analyser.getByteFrequencyData(data);
        setWaveform(Array.from(data.slice(0, 30)).map(v => v / 255));
        animFrameRef.current = requestAnimationFrame(updateWave);
      };
      animFrameRef.current = requestAnimationFrame(updateWave);
    };

    const stopRecording = (): Promise<Blob> => new Promise(resolve => {
      if (!mediaRecorderRef.current) return;
      mediaRecorderRef.current.onstop = () => {
        resolve(new Blob(chunksRef.current, { type: "audio/webm" }));
      };
      mediaRecorderRef.current.stop();
      mediaRecorderRef.current.stream.getTracks().forEach(t => t.stop());
      if (timerRef.current) clearInterval(timerRef.current);
      if (animFrameRef.current) cancelAnimationFrame(animFrameRef.current);
      setIsRecording(false);
      setDuration(0);
      setWaveform([]);
    });

    const cancelRecording = () => {
      if (!mediaRecorderRef.current) return;
      mediaRecorderRef.current.stop();
      mediaRecorderRef.current.stream.getTracks().forEach(t => t.stop());
      if (timerRef.current) clearInterval(timerRef.current);
      if (animFrameRef.current) cancelAnimationFrame(animFrameRef.current);
      setIsRecording(false);
      setDuration(0);
      setWaveform([]);
    };

    return { isRecording, duration, waveform, startRecording, stopRecording, cancelRecording };
  }

=== USE-WEBRTC HOOK ===

  const ICE_SERVERS = [
    { urls: "stun:stun.l.google.com:19302" },
    { urls: "stun:stun1.l.google.com:19302" }
  ];

  export function useWebRTC({ callId, targetUserId, isInitiator, callType }) {
    const pcRef = useRef<RTCPeerConnection | null>(null);
    const [remoteStream, setRemoteStream] = useState<MediaStream | null>(null);
    const [localStream, setLocalStream] = useState<MediaStream | null>(null);
    const [callState, setCallState] = useState<"connecting"|"ringing"|"active"|"ended">("connecting");
    const [isMuted, setIsMuted] = useState(false);
    const [isVideoEnabled, setIsVideoEnabled] = useState(callType === "video");
    
    const init = async () => {
      const constraints = { audio: true, video: callType === "video" ? { facingMode: "user" } : false };
      const stream = await navigator.mediaDevices.getUserMedia(constraints);
      setLocalStream(stream);
      
      const pc = new RTCPeerConnection({ iceServers: ICE_SERVERS });
      pcRef.current = pc;
      
      stream.getTracks().forEach(t => pc.addTrack(t, stream));
      
      pc.ontrack = e => { if (e.streams[0]) setRemoteStream(e.streams[0]); };
      pc.onicecandidate = e => {
        if (e.candidate) {
          echoWs.send(JSON.stringify({ type: "call_signal", targetUserId, signal: { type: "ice-candidate", candidate: e.candidate } }));
        }
      };
      pc.onconnectionstatechange = () => {
        if (pc.connectionState === "connected") setCallState("active");
        if (["disconnected","failed","closed"].includes(pc.connectionState)) setCallState("ended");
      };
      
      if (isInitiator) {
        const offer = await pc.createOffer();
        await pc.setLocalDescription(offer);
        echoWs.send(JSON.stringify({ type: "call_signal", targetUserId, signal: { type: "offer", sdp: offer.sdp } }));
      }
    };
    
    const handleSignal = async (signal: Record<string,unknown>) => {
      const pc = pcRef.current;
      if (!pc) return;
      if (signal.type === "offer") {
        await pc.setRemoteDescription({ type: "offer", sdp: signal.sdp as string });
        const answer = await pc.createAnswer();
        await pc.setLocalDescription(answer);
        echoWs.send(JSON.stringify({ type: "call_signal", targetUserId, signal: { type: "answer", sdp: answer.sdp } }));
        setCallState("active");
      } else if (signal.type === "answer") {
        await pc.setRemoteDescription({ type: "answer", sdp: signal.sdp as string });
      } else if (signal.type === "ice-candidate") {
        await pc.addIceCandidate(signal.candidate as RTCIceCandidateInit);
      }
    };
    
    const toggleMute = () => {
      localStream?.getAudioTracks().forEach(t => { t.enabled = !t.enabled; });
      setIsMuted(v => !v);
    };
    
    const toggleVideo = () => {
      localStream?.getVideoTracks().forEach(t => { t.enabled = !t.enabled; });
      setIsVideoEnabled(v => !v);
    };
    
    const hangup = () => {
      localStream?.getTracks().forEach(t => t.stop());
      pcRef.current?.close();
      setLocalStream(null);
      setRemoteStream(null);
      setCallState("ended");
    };
    
    return { init, handleSignal, hangup, toggleMute, toggleVideo,
             localStream, remoteStream, callState, isMuted, isVideoEnabled };
  }

sw.js:
  const CACHE_NAME = "echo-v1";
  const STATIC_ASSETS = ["/", "/index.html", "/manifest.json"];

  self.addEventListener("install", e => {
    e.waitUntil(caches.open(CACHE_NAME).then(c => c.addAll(STATIC_ASSETS)));
    self.skipWaiting();
  });

  self.addEventListener("activate", e => {
    e.waitUntil(caches.keys().then(keys => Promise.all(
      keys.filter(k => k !== CACHE_NAME).map(k => caches.delete(k))
    )));
    self.clients.claim();
  });

  self.addEventListener("fetch", e => {
    if (e.request.method !== "GET") return;
    if (e.request.url.includes("/api/")) return; // не кэшировать API
    e.respondWith(caches.match(e.request).then(r => r || fetch(e.request)));
  });

  self.addEventListener("push", e => {
    const data = e.data?.json() ?? {};
    e.waitUntil(self.registration.showNotification(data.title ?? "ECHO", {
      body: data.body ?? "Новое сообщение",
      icon: "/icons/icon-192.png",
      badge: "/icons/badge-72.png",
      vibrate: [200, 100, 200],
      data: { url: data.chatUrl ?? "/" },
      actions: [
        { action: "open", title: "Открыть" },
        { action: "dismiss", title: "Закрыть" }
      ]
    }));
  });

  self.addEventListener("notificationclick", e => {
    e.notification.close();
    if (e.action === "dismiss") return;
    e.waitUntil(clients.openWindow(e.notification.data?.url ?? "/"));
  });

export function startBackgroundJobs() {
  // Самоуничтожение сообщений (каждые 10 сек)
  setInterval(async () => {
    const expired = await db.select({ id: messagesTable.id, chatId: messagesTable.chatId })
      .from(messagesTable)
      .where(and(
        lte(messagesTable.selfDestructAt, new Date()),
        eq(messagesTable.isDeleted, false),
        isNotNull(messagesTable.selfDestructAt)
      ));
    
    for (const msg of expired) {
      await db.update(messagesTable).set({ isDeleted: true }).where(eq(messagesTable.id, msg.id));
      broadcastToChat([/* member ids */], { type: "delete_message", messageId: msg.id, chatId: msg.chatId });
    }
  }, 10_000);

  // Напоминания (каждые 30 сек)
  setInterval(async () => {
    const due = await db.select().from(messageRemindersTable)
      .where(and(lte(messageRemindersTable.remindAt, new Date()), eq(messageRemindersTable.isDone, false)));
    
    for (const reminder of due) {
      sendToUser(reminder.userId, { type: "reminder", messageId: reminder.messageId });
      await db.update(messageRemindersTable).set({ isDone: true }).where(eq(messageRemindersTable.id, reminder.id));
    }
  }, 30_000);

  // Истечение Premium подписок (каждый час)
  setInterval(async () => {
    const expired = await db.select({ userId: premiumSubscriptionsTable.userId })
      .from(premiumSubscriptionsTable)
      .where(and(lte(premiumSubscriptionsTable.expiresAt, new Date()), eq(premiumSubscriptionsTable.isActive, true)));
    
    for (const { userId } of expired) {
      await db.update(usersTable).set({ isPremium: false, premiumUntil: null }).where(eq(usersTable.id, userId));
      await db.update(premiumSubscriptionsTable).set({ isActive: false }).where(eq(premiumSubscriptionsTable.userId, userId));
    }
  }, 3_600_000);
}
